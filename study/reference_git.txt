// 변경된 파일들을 변경 전 상태로 바꾸기  // (※ 수정했던 작업이 날아감.)
git checkout .



// 방금 적은 커밋 삭제하기 //
git reset --soft HEAD~1



// 커밋 메세지로 커밋 해시 찾기 // (※ 가장 최근에 한 버전을 가져오려면 'git log -1' 로 적는것이 편하다.)
git log --grep "커밋메세지"



// 커밋 해시를 사용해 체크아웃 // (원하는 버전 가져오기) (※ 브랜치를 사용하고 있을 땐 조심해야함.)
git checkout "커밋 해시"



// 현재 디렉토리의 용량 확인 //
du -sh .



// 현재 디렉토리에 깃 폴더를 제외하고 100MB 넘는 파일 검색 //
find . -path ./.git -prune -o -type f -size +100M -print
